get_git() ## usage get_git tag src dest
{
	cd $CMSSW_BASE/src
        tag=$1 && shift
        src=$1 && shift
        dst=$1 && shift
        wd=$(pwd)

        echo
        echo "Checking out $tag $src into $dst"

        if [[ ! -d $dst ]]; then
         git clone $src $dst
        fi
        cd $dst
        git fetch origin
        git checkout $tag
        
        cd $wd
        echo "done"
        echo
        
        packs="$packs $dst"
}
